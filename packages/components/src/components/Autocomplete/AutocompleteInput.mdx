import { Playground, Props } from 'docz'
import Pane from '../Panel/Panel.js'
import WithState from 'scripts/docz/WithState'

<Playground>
  {() => (
    <WithState>
      {({value, onChange}) => (
        <Pane>
          <AutocompleteInput

            value={value}

            options={[

              'gren',
              'blue',
              'yellow',
              'magenta',
              'white',
              'black',
              'gray',
              'brown',
            ]}
          />
        </Pane>
      )}
    </WithState>
  )}
</Playground>
